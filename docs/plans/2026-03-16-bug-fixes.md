```tsx
'use client'

import { useState, useEffect } from 'react'
import Link from 'next/link'
import { Building2, MapPin, ArrowLeft } from 'lucide-react'

export default function MillDetailPage({ params }: { params: { millId: string } }) {
    const [mill, setMill] = useState<any>(null)
    const [loading, setLoading] = useState(true)
    const [error, setError] = useState<string | null>(null)

    useEffect(() => {
        // Fetch mill details from the mills list and filter by id
        fetch('/api/mills')
            .then(r => r.json())
            .then(d => {
                const found = (d.data ?? []).find((m: any) => m.id === params.millId)
                setMill(found ?? null)
                setLoading(false)
            })
            .catch(() => { setError('Failed to load mill'); setLoading(false) })
    }, [params.millId])

    if (loading) return (
        <div className="flex items-center justify-center h-64">
            <div className="w-6 h-6 rounded-full border-2 border-orange-400 border-t-transparent animate-spin" />
        </div>
    )
    if (error || !mill) return <div className="text-red-500 text-sm p-4">{error ?? 'Mill not found.'}</div>

    return (
        <div className="space-y-6">
            <div className="flex items-center gap-3">
                <Link href="/aggregator/mills" className="text-zinc-400 hover:text-zinc-600 transition">
                    <ArrowLeft size={16} />
                </Link>
                <div>
                    <h1 className="text-xl font-bold text-zinc-900">{mill.name}</h1>
                    <p className="text-sm text-zinc-400 mt-0.5 flex items-center gap-1.5">
                        <span className="font-mono text-xs bg-zinc-100 text-zinc-600 px-1.5 py-0.5 rounded">{mill.code}</span>
                        {mill.location && <><MapPin size={11} />{mill.location}</>}
                    </p>
                </div>
            </div>

            <div className="bg-white rounded-xl border border-zinc-100 shadow-card p-6 space-y-3">
                <h2 className="text-sm font-semibold text-zinc-700">Mill Info</h2>
                <div className="grid grid-cols-2 gap-4 text-sm">
                    <div><span className="text-zinc-400">Country</span><p className="font-medium text-zinc-800">{mill.country ?? '—'}</p></div>
                    <div><span className="text-zinc-400">Status</span><p className="font-medium text-zinc-800">{mill.isActive ? 'Active' : 'Inactive'}</p></div>
                </div>
            </div>

            <div className="bg-white rounded-xl border border-zinc-100 shadow-card p-6">
                <h2 className="text-sm font-semibold text-zinc-700 mb-4">Quick Links</h2>
                <div className="flex gap-3 flex-wrap">
                    <Link href={`/aggregator/mills/${params.millId}/integrations`}
                        className="text-xs font-medium text-orange-500 hover:text-orange-600 underline underline-offset-2">
                        Manage Integrations
                    </Link>
                </div>
            </div>
        </div>
    )
}
```
